create sequence seq_news_id;

CREATE TABLE news (
	id varchar2(100) primary key, 
	title varchar2(255) not null,
	publisher varchar(255),
	author varchar2(100),
	summary varchar(3000),
	imageUrl varchar(255),
	contentUrl varchar(500),
	publishedAt date,
	keyword varchar(255)
	);
